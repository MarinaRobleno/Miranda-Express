CREATE TABLE ROOMS:

CREATE TABLE IF NOT EXISTS rooms(
    id INT AUTO_INCREMENT PRIMARY KEY,
   roomNumber VARCHAR(255) NOT NULL,
   roomType VARCHAR(255) NOT NULL,
    amenities VARCHAR(255),
    price INT NOT NULL,
    offer_price INT,
    status VARCHAR(255)
)ENGINE=INNODB;

CREATE TABLE IF NOT EXISTS rooms(id INT AUTO_INCREMENT PRIMARY KEY, roomNumber VARCHAR(255) NOT NULL, roomType VARCHAR(255) NOT NULL, amenities VARCHAR(255), price INT NOT NULL, offer_price INT, status VARCHAR(255))ENGINE=INNODB;

CREATE TABLE IF NOT EXISTS roomPhotos(
    id INT AUTO_INCREMENT PRIMARY KEY,
    url VARCHAR(255) NOT NULL,
    FOREIGN KEY (id)
        REFERENCES rooms(id)
        ON UPDATE CASCADE
        ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS roomPhotos(id INT AUTO_INCREMENT PRIMARY KEY, url VARCHAR(255) NOT NULL, FOREIGN KEY (id) REFERENCES rooms(id) ON UPDATE CASCADE ON DELETE CASCADE);

CREATE TABLE IF NOT EXISTS bookings(
    id INT AUTO_INCREMENT PRIMARY KEY,
    guest VARCHAR(255) NOT NULL,
    orderDate VARCHAR(255),
    checkIn VARCHAR(255),
    checkOut VARCHAR(255),
    special TEXT,
    status VARCHAR(255)
    FOREIGN KEY (id)
        REFERENCES rooms(id)
        ON UPDATE CASCADE
        ON DELETE CASCADE
) ENGINE=INNODB;

CREATE TABLE IF NOT EXISTS bookings(id INT AUTO_INCREMENT PRIMARY KEY, guest VARCHAR(255) NOT NULL, orderDate VARCHAR(255), checkIn VARCHAR(255), checkOut VARCHAR(255), special TEXT, status VARCHAR(255), FOREIGN KEY (id) REFERENCES rooms(id) ON UPDATE CASCADE ON DELETE SET NULL);

CREATE TABLE IF NOT EXISTS contact(id INT AUTO_INCREMENT PRIMARY KEY, photo VARCHAR(255), date VARCHAR(255), customer VARCHAR(255), mail VARCHAR(255), phone VARCHAR(255), subject VARCHAR(255), comment TEXT)ENGINE=INNODB;


CREATE TABLE IF NOT EXISTS users(
    id INT AUTO_INCREMENT PRIMARY KEY,
    photo VARCHAR(255),
    name VARCHAR(255),
    mail VARCHAR(255),
    job VARCHAR(255),
    phone VARCHAR(255),
    status VARCHAR(255),
    startDate DATE,
    endDate DATE
)

CREATE TABLE IF NOT EXISTS users(id INT AUTO_INCREMENT PRIMARY KEY, photo VARCHAR(255), name VARCHAR(255), mail VARCHAR(255), job VARCHAR(255), phone VARCHAR(255), status VARCHAR(255), startDate VARCHAR(255), endDate VARCHAR(255))
